;; --- Vector and Matrix Utility Functions ---

 

;; Returns the angle between two vectors in radians

{angle(A, B) acos(dot(A, B) / (norm(A) * norm(B)))}

 

;; Returns the Euclidean distance between two points (vectors)

{dist(A, B) norm(A - B)}

 

;; Returns the unit vector (direction) of A

{unit(A) A / norm(A)}

 

;; Solves a system of linear equations A*X = B (returns vector X)

{solve_lin(A, B) !A * B}

 

;; Returns the projection of vector A onto vector B

{proj(A, B) (dot(A, B) / dot(B, B)) * B}

 

;; Linear interpolation between A and B (t from 0 to 1)

{lerp(A, B, t) A + t * (B - A)}

 

;; Returns the scalar triple product of three 3D vectors

{triple(A, B, C) dot(A, cross(B, C))}

 

;; --- Coordinate Transformations ---

 

;; Returns a 2D rotation matrix for a given angle alpha

{rot2(a) [(cos(a), -sin(a)); (sin(a), cos(a))]}
